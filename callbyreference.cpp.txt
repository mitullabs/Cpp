//Program to swap two numbers using call by reference
#include<iostream>  
using namespace std; 

void swap(int *x, int *y)  
{  
 int swap;  
 swap=*x;  
 *x=*y;  
 *y=swap;  
}  

int main()   
{    
 int x=500, y=100;
 cout<<"Before Swapping\n";  
 cout<<"Value of x is: "<<x<<endl;  
 cout<<"Value of y is: "<<y<<endl;
 cout<<"After Swapping\n";   
 swap(&x, &y);  // passing value to function  
 cout<<"Value of x is: "<<x<<endl;  
 cout<<"Value of y is: "<<y<<endl;  
 return 0;  
}
