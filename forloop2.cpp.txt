// C++ program to demonstrate the use of Nested for loop. 
#include <iostream> 
using namespace std; 
  
int main() 
{ 
    for (int i = 0; i < 4; i++) 
    { 
        for (int j = 0; j < 4; j++) 
        { 
            cout << "*" << " "; 
        } 
        cout << endl; 
    } 
return 0;
}
