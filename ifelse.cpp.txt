// Program to check whether the given number is positive or negative
#include <iostream>
using namespace std;

int main() 
{
  int number;
  cout << "Enter a number: ";
  cin >> number;
  if (number >= 0) 
  {
    cout << "You entered a positive number: " << number << endl;
  }
  else 
  {
    cout << "You entered a negative number: " << number << endl;
  }
  cout << "Checking is done!";
  return 0;
}
