/*C++ Program to evaluate largest of the three numbers
  using nested if else*/
#include <iostream>
using namespace std;

int main()
{
    int a = 10;
    int b = 2;
    int c = 6;
    if (a < b) 
    {
        if (c < b) 
        {
            cout<<b<<" is the greatest"<<"\n";
        }
        else 
        {
            cout<<c<<" is the greatest"<<"\n";
        }
    }
    else 
    {
        if (c < a) 
        {
            cout<<a<<" is the greatest"<<"\n";
        }
        else 
        {
            cout<<c<<" is the greatest"<<"\n";
        }
    }
    return 0;
}
